הקוד הבא משתמש ב-Python עם ספריית OpenCV. הוא מבצע זיהוי קצוות (Edges) וזיהוי קווים (Hough Lines) כדי "לחלץ" את מבנה המסלולים מהתמונה שצירפת:

import cv2
import numpy as np
import matplotlib.pyplot as plt

def image_to_vector_map(image_path):
    # 1. טעינת התמונה
    img = cv2.imread(image_path)
    if img is None:
        print("לא ניתן לטעון את התמונה")
        return
    
    # המרה לגווני אפור וטשטוש להפחתת רעשים
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    blurred = cv2.GaussianBlur(gray, (5, 5), 0)

    # 2. זיהוי קצוות (Canny Edge Detection)
    edges = cv2.Canny(blurred, 50, 150, apertureSize=3)

    # 3. זיהוי קווים (Hough Line Transform) - זה מזהה את המסלולים
    # הפרמטרים מכוונים לזהות קווים ארוכים יחסית (מסלולים)
    lines = cv2.HoughLinesP(edges, 1, np.pi/180, threshold=100, minLineLength=100, maxLineGap=10)

    # יצירת תמונה לבנה עליה נצייר את המפה הוקטורית
    vector_map = np.ones_like(img) * 255 

    # 4. ציור המסלולים שזוהו
    if lines is not None:
        for line in lines:
            x1, y1, x2, y2 = line[0]
            # מצייר את המסלולים באפור כהה (כמו מפה וקטורית)
            cv2.line(vector_map, (x1, y1), (x2, y2), (50, 50, 50), 5)

    # 5. זיהוי דת"קים (באמצעות זיהוי צורות קטנות/ריבועים)
    # אנחנו מחפשים קונטורים (מתווים) סגורים
    contours, _ = cv2.findContours(edges, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)
    
    for cnt in contours:
        area = cv2.contourArea(cnt)
        # פילטר לפי גודל - דת"קים הם בדרך כלל בגודל מסוים בתמונה
        if 100 < area < 1000: 
            x, y, w, h = cv2.boundingRect(cnt)
            # ציור ריבוע אדום עבור כל דת"ק פוטנציאלי
            cv2.rectangle(vector_map, (x, y), (x + w, y + h), (0, 0, 255), 2)

    # הצגת התוצאה
    plt.figure(figsize=(12, 8))
    plt.subplot(1, 2, 1)
    plt.title("Original Image")
    plt.imshow(cv2.cvtColor(img, cv2.COLOR_BGR2RGB))
    plt.axis('off')

    plt.subplot(1, 2, 2)
    plt.title("Vector Map Extraction")
    plt.imshow(cv2.cvtColor(vector_map, cv2.COLOR_BGR2RGB))
    plt.axis('off')
    
    plt.tight_layout()
    plt.show()

# הרצת הפונקציה על הקובץ שלך
image_to_vector_map('רמת דויד.jpg')

הסבר על מה שהקוד עושה:


Canny Edge Detection: הוא "מנקה" את התמונה ומשאיר רק את קווי המתאר של המסלולים והמבנים.

HoughLinesP: זהו אלגוריתם מתמטי שמזהה קווים ישרים. הוא מוצא את המסלולים הראשיים (09/27 ו-15/33) ומתעלם מרקע חקלאי או רעשים.

Contour Analysis: הקוד סורק צורות גיאומטריות סגורות. הגדרתי פילטר שטח (area) כדי שיזהה מבנים קטנים יחסית כמו דת"קים ויסמן אותם בריבועים.
